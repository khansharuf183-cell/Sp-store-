<script>
const products = [
  {name:"Wireless Earbuds",price:499,img:"https://via.placeholder.com/200"},
  {name:"Smart Watch",price:899,img:"https://via.placeholder.com/200"},
  {name:"Men T-Shirt",price:299,img:"https://via.placeholder.com/200"},
  {name:"Bluetooth Speaker",price:699,img:"https://via.placeholder.com/200"},
  {name:"Shoes",price:999,img:"https://via.placeholder.com/200"}
];

let cart = [];

function loadProducts(){
  const list = document.getElementById("productList");
  list.innerHTML = "";
  products.forEach((p,i)=>{
    list.innerHTML += `
      <div class="card">
        <img src="${p.img}">
        <h4>${p.name}</h4>
        <p class="price">₹${p.price}</p>
        <button onclick="addToCart(${i})">Add to Cart</button>
      </div>
    `;
  });
}

function addToCart(i){
  cart.push(products[i]);
  document.getElementById("count").innerText = cart.length;
  alert("Added to cart ✅");
}

function orderNow(){
  if(cart.length === 0){
    alert("Cart is empty!");
    return;
  }

  let total = 0;
  let text = "🛒 Order Details:%0A";

  cart.forEach(p=>{
    text += p.name + " - ₹" + p.price + "%0A";
    total += p.price;
  });

  text += "Total: ₹" + total;

  let phone = "918134834380"; // 🔴 apna number dalna
  window.open(`https://wa.me/${phone}?text=${text}`);
}

loadProducts();
</script>
